Add this cell to your Kaggle/Colab notebook after training and saving brain_tumor_model.keras. It creates a Flask API and exposes it publicly via ngrok.

!pip install flask flask-cors pyngrok pillow -q

from flask import Flask, request, jsonify
from flask_cors import CORS
from pyngrok import ngrok
import tensorflow as tf
import numpy as np
from PIL import Image
import base64, io, re

app = Flask(__name__)
CORS(app)

model = tf.keras.models.load_model("brain_tumor_model.keras")
CLASS_NAMES = ["glioma", "meningioma", "notumor", "pituitary"]

@app.route("/predict", methods=["POST"])
def predict():
    data = request.get_json()
    if not data or "image" not in data:
        return jsonify({"error": "No image provided"}), 400
    # Strip data URL header: "data:image/png;base64,..."
    b64 = re.sub(r"^data:image/[^;]+;base64,", "", data["image"])
    img_bytes = base64.b64decode(b64)
    img = Image.open(io.BytesIO(img_bytes)).convert("RGB")
    img = img.resize((256, 256))
    arr = np.array(img) / 255.0
    arr = np.expand_dims(arr, axis=0)
    preds = model.predict(arr)[0]
    idx = int(np.argmax(preds))
    return jsonify({
        "predicted_class": CLASS_NAMES[idx],
        "confidence": float(preds[idx]),
        "probabilities": {CLASS_NAMES[i]: float(preds[i]) for i in range(4)}
    })

public_url = ngrok.connect(5000)
print("BrainDX API live at:", public_url, "/predict")

app.run(port=5000)
